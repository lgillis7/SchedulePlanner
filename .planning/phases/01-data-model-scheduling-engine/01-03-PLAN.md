  Promise<{ project: Project; tasks: RawTask[]; dependencies: Dependency[]; owners: Owner[] }>

// Task CRUD
export async function createTask(client, task: { projectId: string; title: string; parentTaskId?: string; tierDepth: number; sortOrder: number }): Promise<RawTask>
export async function updateTask(client, taskId: string, updates: Partial<{title, desiredStartDate, durationDays, completionPct, ownerId, parentTaskId, tierDepth, sortOrder, notes}>): Promise<RawTask>
export async function deleteTask(client, taskId: string): Promise<void>

// Owner CRUD
export async function createOwner(client, owner: { projectId: string; name: string; color: string; contactInfo?: string }): Promise<Owner>

export async function deleteOwner(client, ownerId: string): Promise<void>

// Dependency CRUD
export async function addDependency(client, dep: { projectId: string; upstreamTaskId: string; downstreamTaskId: string }): Promise<Dependency>
export async function removeDependency(client, depId: string): Promise<void>

// Project
export async function updateProject(client, projectId: string, updates: Partial<{name, includeWeekends}>): Promise<Project>

Each function handles Supabase errors by throwing with a clear message. Use the typed `Database` type from `@/types/database` if generated, or use `any` with a TODO to generate types later.

Main container component. Renders the hierarchical task list as a table/list.

- Receives `schedule: ComputedTask[]`, `owners: Owner[]`, `dependencies: Dependency[]`, and mutation callbacks as props
- Sorts tasks by `sortOrder` and nests them visually by `parentTaskId` / `tierDepth`
- Columns: Line # (sortOrder), Title (tier-formatted with indent), Owner (color dot + name), Desired Start, Duration, Required Start (computed), End Date (computed), Completion %, Dependencies (upstream line numbers), Actions (edit/delete)
- "Add Task" button at bottom (creates at tier 0 by default) and "Add Subtask" in each row's actions
- Uses `tierStyles()` and `tierIndent()` from formatting.ts for hierarchy visual
- Line numbers are displayed as `sortOrder` values (stable IDs for dependency input, per DEP-02)

**Create `src/components/task-list/TaskRow.tsx`:**

Single row in the task table.

- Displays one ComputedTask with tier formatting
- Shows owner color as a small colored circle/badge next to owner name
- Computed dates (requiredStartDate, endDate) displayed in lighter color to distinguish from user inputs
- If task has no owner, show "Unassigned" in muted text
- Inline completion % display (just the number, editable in TaskEditor)
- Dependencies column: comma-separated upstream line numbers
- Actions: Edit (opens TaskEditor), Delete (with confirmation dialog), Add Subtask

**Create `src/components/task-list/TaskEditor.tsx`:**

Slide-over panel or dialog for editing a task. Uses shadcn/ui Dialog.

Fields:
- Title: text Input
- Desired Start Date: shadcn/ui Calendar in a Popover (TASK-04) -- displays formatted date, click opens calendar
- Duration (days): number Input, step 0.5, min 0.1 (TASK-05)
- Completion %: number Input, min 0, max 100, step 5 (PROG-01)
- Owner: Select dropdown populated from owners list, with "Unassigned" option (OWN-03)
- Dependencies: text Input accepting comma-separated line numbers (DEP-02). On save, resolve line numbers to task UUIDs via sortOrder lookup. Show validation error if line number doesn't exist or would create a cycle.
- Parent Task: Select dropdown of existing tasks (for re-parenting), filtered to prevent setting self as parent
- Notes: textarea

On save: call appropriate `updateTask` mutation, then `refetch()`. On dependency change: call `addDependency`/`removeDependency` as needed.

**Dependency validation on save:** Before persisting new dependencies, run `detectCycle` from the scheduling engine with the proposed new dependency added. If cycle detected, show toast error with cycle task names. Do NOT persist the dependency.
