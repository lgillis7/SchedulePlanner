Create `src/lib/supabase/server.ts` -- server Supabase client using `@supabase/ssr` `createServerClient` with Next.js cookies:
```typescript
import { createServerClient } from '@supabase/ssr';
import { cookies } from 'next/headers';
import type { Database } from '@/types/database';

export async function createClient() {
  const cookieStore = await cookies();
  return createServerClient<Database>(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll() { return cookieStore.getAll(); },
        setAll(cookiesToSet) {
          cookiesToSet.forEach(({ name, value, options }) =>
            cookieStore.set(name, value, options)
          );
        },
      },
    }
  );
}
```

Create `supabase/migrations/00001_initial_schema.sql` with the full schema from the research document. Tables:

1. **projects** -- id (UUID PK), name (TEXT NOT NULL), include_weekends (BOOLEAN DEFAULT false), created_at, updated_at. Enable RLS with permissive policy.

2. **owners** -- id (UUID PK), project_id (FK to projects ON DELETE CASCADE, NOT NULL), name (TEXT NOT NULL), contact_info (TEXT), color (TEXT NOT NULL DEFAULT '#3B82F6'), sort_order (INTEGER DEFAULT 0), created_at. Enable RLS with permissive policy.

3. **tasks** -- id (UUID PK), project_id (FK to projects ON DELETE CASCADE, NOT NULL), parent_task_id (FK to tasks ON DELETE CASCADE, nullable), owner_id (FK to owners ON DELETE SET NULL, nullable), title (TEXT NOT NULL), tier_depth (INTEGER NOT NULL DEFAULT 0, CHECK 0-3), sort_order (INTEGER NOT NULL DEFAULT 0), desired_start_date (DATE NOT NULL DEFAULT CURRENT_DATE), duration_days (FLOAT NOT NULL DEFAULT 1.0, CHECK > 0), completion_pct (FLOAT DEFAULT 0.0, CHECK 0-100), notes (TEXT), created_at, updated_at. Enable RLS with permissive policy.

4. **task_dependencies** -- id (UUID PK), project_id (FK to projects ON DELETE CASCADE, NOT NULL), upstream_task_id (FK to tasks ON DELETE CASCADE, NOT NULL), downstream_task_id (FK to tasks ON DELETE CASCADE, NOT NULL), dependency_type (TEXT DEFAULT 'finish-to-start'), created_at. UNIQUE(upstream_task_id, downstream_task_id). CHECK(upstream != downstream). Enable RLS with permissive policy.

5. **checkpoints** -- id (UUID PK), project_id (FK to projects ON DELETE CASCADE, NOT NULL), captured_at (TIMESTAMPTZ DEFAULT now()), total_work_days (FLOAT NOT NULL), completed_work_days (FLOAT NOT NULL), notes (TEXT). Enable RLS with permissive policy.

Add indexes: tasks(project_id), tasks(parent_task_id), tasks(project_id, sort_order), owners(project_id), task_dependencies(project_id), task_dependencies(upstream_task_id), task_dependencies(downstream_task_id).
